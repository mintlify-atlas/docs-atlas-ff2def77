<div className="mt-16 mb-16 max-w-5xl mx-auto px-6">
  <h2 className="text-3xl font-bold text-gray-900 dark:text-white mb-4">
    Quick Start
  </h2>
  <p className="text-lg text-gray-600 dark:text-gray-400 mb-8">
    Get WalletSync up and running in minutes
  </p>

  <Steps>
    <Step title="Clone and build the project">
      Clone the repository and build the application using Maven:
      
      ```bash
      git clone https://github.com/alemadorran/walletsync.git
      cd walletsync
      ./mvnw clean package
      ```
    </Step>

    <Step title="Configure the database">
      Set up PostgreSQL and update `application.properties`:
      
      ```properties
      spring.datasource.url=jdbc:postgresql://localhost:5432/walletsync_db
      spring.datasource.username=postgres
      spring.datasource.password=your_password
      ```
      
      <Note>The application will automatically create database tables on startup using Hibernate DDL auto-update.</Note>
    </Step>

    <Step title="Start the application">
      Run the Spring Boot application:
      
      ```bash
      ./mvnw spring-boot:run
      ```
      
      The API will be available at `http://localhost:8080`
    </Step>

    <Step title="Make your first API call">
      Test the API by creating a category:
      
      ```bash cURL
      curl -X POST http://localhost:8080/api/categories \
        -H "Content-Type: application/json" \
        -d '{
          "categoryName": "Groceries",
          "categoryType": "EXPENSE"
        }'
      ```
      
      <Accordion title="Example response">
        ```json
        {
          "id": 1,
          "categoryName": "Groceries",
          "categoryType": "EXPENSE"
        }
        ```
      </Accordion>
    </Step>
  </Steps>
</div>
